    #inXæ˜¯ä¸ªå‘é‡ï¼Œè€Œdatasetæ˜¯ä¸ªçŸ©é˜µï¼Œä¸¤è€…ä¹‹é—´è¦è¿›è¡Œç›¸å‡çš„è¿ç®—ï¼Œéœ€è¦æŠŠè¿™ä¸ªå‘é‡ä¹Ÿè¡¥æˆä¸€ä¸ªå’Œdatasetæœ‰ç›¸åŒè¡Œæ•°åˆ—æ•°çš„çŸ©é˜µï¼Œ
    #tile()çš„ç¬¬äºŒä¸ªå‚æ•°ï¼Œå°±æ˜¯(datasetsize,1)ï¼Œè¿™ä¸ªå‚æ•°çš„æ„æ€å°±æ˜¯æŠŠinXè¡¥æˆæœ‰datasetsizeè¡Œæ•°çš„çŸ©é˜µã€‚
    #å‡å¦‚inXæ˜¯ï¼ˆ1ï¼Œ2ï¼‰ï¼Œdatasetsize =3ï¼Œé‚£ä¹ˆç»è¿‡tile()è½¬æ¢åäº§ç”Ÿäº†ä¸€ä¸ªè¿™æ ·çš„çŸ©é˜µï¼ˆ[1,2],[1,2],[1,2]ï¼‰
    sqDiffMat = diffMat**2  #å¹³æ–¹

    # sqdiffMatæ˜¯([1,2],[0,1],[3,4])ï¼Œaxisè¿™ä¸ªå‚æ•°å½±å“äº†å¯¹çŸ©é˜µæ±‚å’Œæ—¶å€™çš„é¡ºåºï¼Œaxis=0æ˜¯æŒ‰ç…§åˆ—æ±‚å’Œï¼Œç»“æœä¸º([3.1.7])
    # axis=1æ˜¯æŒ‰ç…§è¡Œè¿›è¡Œæ±‚å’Œï¼Œç»“æœæ˜¯([4,7])ã€‚
    distances = sqDistances**0.5    #å¼€æ–¹ï¼Œå¾—åˆ°æ¬§æ°è·ç¦»
    sortedDistIndicies = distances.argsort()     #æŠŠå‘é‡ä¸­æ¯ä¸ªå…ƒç´ è¿›è¡Œæ’åºï¼Œç»“æœæ˜¯å…ƒç´ çš„ç´¢å¼•å½¢æˆçš„å‘é‡
    #ä¾‹å­distance([1,4,3])ï¼Œç»è¿‡distance.argsort()ä¹‹åçš„ç»“æœæ˜¯([0,2,1]

    #æŠ•ç¥¨è¿‡ç¨‹ï¼Œå°±æ˜¯ç»Ÿè®¡å‰kä¸ªæœ€è¿‘çš„æ ·æœ¬æ‰€å±ç±»åˆ«åŒ…å«çš„æ ·æœ¬ä¸ªæ•°

        # voteIlabel = labels[index]æ˜¯æ ·æœ¬indexå¯¹åº”çš„åˆ†ç±»ç»“æœ('A' or 'B')
        voteIlabel = labels[sortedDistIndicies[i]]
        # classCount.get(voteIlabel, 0)è¿”å›voteIlabelçš„å€¼ï¼Œå¦‚æœä¸å­˜åœ¨ï¼Œåˆ™è¿”å›0
        # ç„¶åå°†ç¥¨æ•°å¢1
        classCount[voteIlabel] = classCount.get(voteIlabel,0) + 1
    # æŠŠåˆ†ç±»ç»“æœè¿›è¡Œæ’åºï¼Œç„¶åè¿”å›å¾—ç¥¨æ•°æœ€å¤šçš„åˆ†ç±»ç»“æœ
    # key=operator.itemgetter(1)çš„æ„æ€æ˜¯æŒ‰ç…§å­—å…¸é‡Œçš„ç¬¬ä¸€ä¸ªæ’åº
    #ä¾‹å­a = [1, 2, 3]ï¼Œb = operator.itemgetter(1)ï¼Œb(a)è¿”å›ä¸º2

    # reverse=Trueæ˜¯é™åºæ’åº
    sortedClassCount = sorted(classCount.items(), key=operator.itemgetter(1), reverse=True)

def file2matrix(filename):

def autoNorm(dataSet):
    minVals = dataSet.min(0)#ä¸€ç»´æ•°ç»„ï¼Œå€¼ä¸ºå„é¡¹ç‰¹å¾ï¼ˆåˆ—ï¼‰ä¸­çš„æœ€å°å€¼ã€‚å‚æ•°0ä½¿å¾—å‡½æ•°ä»åˆ—ä¸­é€‰å–æœ€å°å€¼
    #print(minVals)
    maxVals = dataSet.max(0)
    #print(maxVals)
    ranges = maxVals - minVals
    normDataSet = zeros(shape(dataSet)) #åˆ›å»ºä¸æ ·æœ¬é›†ä¸€æ ·å¤§å°çš„é›¶çŸ©é˜µ
    #print(normDataSet)
    m = dataSet.shape[0]#dataSetçš„è¡Œæ•°
    normDataSet = dataSet - tile(minVals, (m,1))#çŸ©é˜µä¸­æ‰€æœ‰çš„å€¼å‡å»æœ€å°å€¼
    #tileå°†åŸæ¥çš„ä¸€ä¸ªæ•°ç»„minValsï¼Œæ‰©å……æˆäº†mè¡Œ1åˆ—çš„æ•°ç»„
    normDataSet = normDataSet/tile(ranges, (m,1))   #çŸ©é˜µä¸­æ‰€æœ‰çš„å€¼é™¤ä»¥æœ€å¤§å–å€¼èŒƒå›´è¿›è¡Œå½’ä¸€åŒ–
    return normDataSet, ranges, minVals
