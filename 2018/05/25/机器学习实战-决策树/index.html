def calcShannonEnt(dataSet):

    labelCounts = {}            #åˆ›å»ºç©ºå­—å…¸

        currentLabel = featVec[-1]  #è·å–ç‰¹å¾å‘é‡æœ€åä¸€åˆ—çš„æ ‡ç­¾
        # æ£€æµ‹å­—å…¸çš„å…³é”®å­—keyä¸­æ˜¯å¦å­˜åœ¨è¯¥æ ‡ç­¾ï¼Œå¦‚æœä¸å­˜åœ¨keys()å…³é”®å­—ï¼Œå°†å½“å‰æ ‡ç­¾/0é”®å€¼å¯¹å­˜å…¥å­—å…¸ä¸­,å¹¶èµ‹å€¼ä¸º0
        #print(labelCounts.keys())

        #print("%s="%currentLabel,labelCounts[currentLabel])

        #print("%s="%labelCounts[key],shannonEnt)

def chooseBestFeatureToSplit(dataSet):

    bestInfoGain = 0.0;bestFeature = -1
    for i in range(numFeatures):

        #print(featList)
        uniqueVals = set(featList)       #ç¡®å®šæŸä¸€ç‰¹å¾ä¸‹æ‰€æœ‰å¯èƒ½çš„å–å€¼,seté›†åˆç±»å‹ä¸­çš„æ¯ä¸ªå€¼äº’ä¸ç›¸åŒ
        #print(uniqueVals)
        newEntropy = 0.0
        for value in uniqueVals:        #è®¡ç®—æ¯ç§åˆ’åˆ†æ–¹å¼çš„ä¿¡æ¯ç†µ
            subDataSet = splitDataSet(dataSet, i, value)        #æŠ½å–è¯¥ç‰¹å¾çš„æ¯ä¸ªå–å€¼ä¸‹å…¶ä»–ç‰¹å¾çš„å€¼ç»„æˆæ–°çš„å­æ•°æ®é›†
            prob = len(subDataSet)/float(len(dataSet))      #è®¡ç®—è¯¥ç‰¹å¾ä¸‹çš„æ¯ä¸€ä¸ªå–å€¼å¯¹åº”çš„æ¦‚ç‡ï¼ˆæˆ–è€…è¯´æ‰€å çš„æ¯”é‡ï¼‰
            newEntropy += prob * calcShannonEnt(subDataSet)     #è®¡ç®—è¯¥ç‰¹å¾ä¸‹æ¯ä¸€ä¸ªå–å€¼çš„å­æ•°æ®é›†çš„ä¿¡æ¯ç†µï¼Œå¹¶æ±‚å’Œ
        infoGain = baseEntropy - newEntropy     #è®¡ç®—æ¯ä¸ªç‰¹å¾çš„ä¿¡æ¯å¢ç›Š

        if (infoGain > bestInfoGain):
            bestInfoGain = infoGain
            bestFeature = i
            #print("ç¬¬%dä¸ªç‰¹å¾çš„ä¿¡æ¯å¢ç›Šæœ€å¤§ï¼Œæ‰€ä»¥é€‰æ‹©å®ƒä½œä¸ºåˆ’åˆ†çš„ä¾æ®ï¼Œå…¶ç‰¹å¾çš„å–å€¼ä¸º%s,å¯¹åº”çš„ä¿¡æ¯å¢ç›Šå€¼æ˜¯%f"%((i+1),uniqueVals,infoGain))

def majorityCnt(classList):
    classCount={}
    for vote in classList:
        if vote not in classCount.keys(): classCount[vote] = 0
        classCount[vote] += 1
    sortedClassCount = sorted(classCount.iteritems(), key=operator.itemgetter(1), reverse=True)
    return sortedClassCount[0][0]
